  90def oval(center=(0, 0), radius_x=25, radius_y=50, color="hotpink", outline="", tag="", **kwargs):
  91    """
  92    A reporter function that draws an oval.
  93    Args:
  94        center (`tuple`): A coordinate representing the center of the shape.
  95        radius_x (`int`): Specifies the oval's radius on the x-axis.
  96        radius_y (`int`): Specifies the oval's radius on the y-axis.
  97        color (`str`): What color to draw the shape.
  98        outline (`str`): What color should the border of the shape be.
  99        tag (`str`): The tag to assign to the shape.
 100
 101    Returns:
 102         `Shape`: The oval that was created.
 103    """
 104    x = center[0]
 105    y = center[1]
 106    x0, y0, x1, y1 = (x - radius_x, y - radius_y, x + radius_x, y + radius_y)
 107    steps = 100
 108    # major and minor axes
 109    a = (x1 - x0) / 2.0
 110    b = (y1 - y0) / 2.0
 111    # center
 112    xc = x0 + a
 113    yc = y0 + b
 114    point_list = []
 115    # create the oval as a list of points
 116    for i in range(steps):
 117        # Calculate the angle for this step
 118        theta = (pi * 2) * (float(i) / steps)
 119        x = a * cos(theta)
 120        y = b * sin(theta)
 121        point_list.append(round(x + xc))
 122        point_list.append(round(y + yc))
 123        
 124    return _a_canvas.create_polygon(point_list, fill=color, tags=tag, outline=outline, **kwargs)

 208def star(center=(0, 0), radius=50, color="hotpink", outer_radius=75, points=5, outline="", tag="", **kwargs):
 209    """
 210    A reporter function that draws a star.
 211    Args:
 212        center (`tuple`): A coordinate representing the center of the shape.
 213        radius (`int`): Specifies the radius of the inside part of the star.
 214        color (`str`): Specifies the color of the star.
 215        outer_radius (`int`): Specifies the radius of the outside part of the star.
 216        points (`int`): Specifies the number of points for the star.
 217        outline (`str`): What color should the border of the shape be.
 218        tag (`str`): The tag to assign to the shape.
 219
 220    Returns:
 221         `Shape`: The star that was created.
 222    """
 223    arc_segment = 360 / points
 224    vertices = []
 225    for i in range(points):
 226        inner_point = (
 227            radius * cos(radians(arc_segment * i)) + center[0],
 228            -1 * radius * sin(radians(arc_segment * i)) + center[1],
 229        )
 230        vertices.append(inner_point)
 231        outer_point = (
 232            outer_radius * cos(radians(arc_segment * i + arc_segment / 2)) + center[0],
 233            -1 * outer_radius * sin(radians(arc_segment * i + arc_segment / 2))
 234            + center[1],
 235        )
 236        vertices.append(outer_point)
 237    return polygon(vertices, color=color, tag=tag, outline=outline, **kwargs)

 257def spiral(center=(0, 0), width=100, roughness=0.01, start=0, spirals=5, line_width=1, tag="", **kwargs):
 258    """
 259    A reporter function that draws a spiral.
 260    Args:
 261        center (`tuple`): A coordinate representing the center of the shape.
 262        width (`int`): Specifies the total width of the spiral.
 263        roughness (`float`): Controls how spiral-y the shape is (lower is less spiral-y)
 264        start (`int`): Where on the spiral to start drawing.
 265        spirals (`int`): How many loops to draw.
 266        line_width (`int`): How wide for the line to be drawn.
 267        tag (`str`): The tag to assign to the shape.
 268
 269    Returns:
 270         `Shape`: The spiral that was created.
 271    """
 272    theta = 0.0
 273    r = start
 274    all_points = []
 275    prev_pos = _polar_to_cartesian(r, theta)
 276    distance = width / 4 / pi / spirals 
 277    all_points.append((prev_pos[0] + center[0], prev_pos[1] + center[1]))
 278    while theta < 2 * spirals * pi:
 279        theta += roughness
 280        r = start + distance*theta
 281        pos = _polar_to_cartesian(r, theta)
 282        all_points.append((pos[0] + center[0], pos[1] + center[1]))
 283
 284    return arc(points=all_points, width=line_width, tag=tag, **kwargs)

 321def move_to(tag, to, anchor="center"):
 322    """
 323    Move the given tagged item to a particular `point` maintaining some `anchor`.
 324    Note: this is NOT the same as the `move` function which moves an object by a specific amount.
 325    
 326    Args:
 327        tag (Shape or `str`): the shape (or shapes) to move
 328        to (`tuple`): the `(x, y)` coordinate to which you wish to move the tagged object
 329        anchor (`str`): which point on the shape do you want to move toward the given tuple. You can
 330            use either `"center"` (default), `"top_left"`, `"top_right"`, `"bottom_left"`, or `"bottom_right"`.
 331    """
 332    anchor_options = ["center", "top_left", "top_right", "bottom_left", "bottom_right"]
 333    if anchor not in anchor_options:
 334        raise ValueError("The anchor input must be one of " + str(anchor_options) + " but instead we found " + str(anchor))
 335
 336    outline = _get_outline(tag)
 337    delta_x = 0
 338    delta_y = 0
 339    
 340    if anchor == "top_left":
 341        delta_x = to[0] - outline['left']
 342        delta_y = to[1] - outline['top'] 
 343    elif anchor == "top_right":
 344        delta_x = to[0] - outline['right']
 345        delta_y = to[1] - outline['top']
 346    elif anchor == "bottom_right":
 347        delta_x = to[0] - outline['right']
 348        delta_y = to[1] - outline['bottom']
 349    elif anchor == "bottom_left":
 350        delta_x = to[0] - outline['left']
 351        delta_y = to[1] - outline['bottom']
 352    elif anchor == "center":
 353        delta_x = to[0] - outline['center'][0]
 354        delta_y = to[1] - outline['center'][1]
 355    
 356    _a_canvas.move(tag, delta_x, delta_y)

 399def align(shape1, shape2, via="middle", offset_x = 0, offset_y = 0):
 400    """
 401    A reporter function that aligns `shape1` with `shape2`. It does this by moving `shape1` to align with
 402    whatever property of `shape2` is selected with the `via` input.
 403    
 404    Args:
 405        shape1 (`Shape` or Tag): The first shape to use.
 406        shape2 (`Shape` or Tag): The second shape to use.
 407        via (`str`): Has to be one of, the following options: `"center"` (horizontal center), 
 408            `"middle"` (vertical center), `"top"`, `"bottom"`, `"left"`, or `"right"`
 409        offset_x (`int`): How much to shift in the x-axis after alignment
 410        offset_y (`int`): How much to shift in the y-axis after alignment
 411
 412    Returns:
 413        `Shape`: The modified shape1.
 414    """
 415    via_options = ["center", "middle", "top", "bottom", "left", "right"]
 416    if via not in via_options:
 417        raise ValueError("The via input must be one of " + str(via_options) + " but instead we found " + str(via))
 418
 419    outline1 = _get_outline(shape1)
 420    outline2 = _get_outline(shape2)
 421    
 422    if via == "center":
 423        _a_canvas.move(
 424            shape1,
 425            (outline2['center'][0] - outline1['center'][0]) + offset_x, offset_y)
 426        
 427    elif via == "middle":
 428        _a_canvas.move(
 429            shape1,
 430            offset_x, (outline2['center'][1] - outline1['center'][1]) + offset_y)
 431        
 432    elif via == "top":
 433        _a_canvas.move(
 434            shape1,
 435            offset_x, (outline2['top'] - outline1['top']) + offset_y)
 436        
 437    elif via == "bottom":
 438        _a_canvas.move(
 439            shape1,
 440            offset_x, (outline2['bottom'] - outline1['bottom']) + offset_y)
 441        
 442    elif via == "left":
 443        _a_canvas.move(
 444            shape1,
 445            (outline2['left'] - outline1['left']) + offset_x, offset_y)
 446        
 447    elif via == "right":
 448        _a_canvas.move(
 449            shape1,
 450            (outline2['right'] - outline1['right']) + offset_x, offset_y)
 451        
 452    return shape1

 636def rotate(shape, degrees=5, origin=None):
 637    """
 638    A reporter function that takes a shape and rotates it by a specified amount around a specified point.
 639    It does this by interpolating a polygon around the shape and calculating the shifts of individual
 640    points on the edge of the polygon.
 641
 642    Args:
 643        shape (`Shape` or Tag): The shape to rotate.
 644        degrees (`int`): The number of degrees to rotate the shape.
 645        origin (`tuple`): An `(x,y)` coordinate about which to perform the rotation. Defaults to the center
 646            of the given shape.
 647
 648    Returns:
 649        `Shape`: The modified shape.  
 650    """
 651    if origin is None:
 652        origin = get_center(shape)
 653
 654    theta = radians(degrees)
 655    ox, oy = origin
 656
 657    all_shapes = _a_canvas.find_withtag(shape)
 658
 659    for a_shape in all_shapes:
 660        coords = _a_canvas.coords(a_shape)
 661        # update coordinates:
 662        for i in range(0, len(coords), 2):
 663            px, py = coords[i], coords[i + 1]
 664            qx = cos(theta) * (px - ox) - sin(theta) * (py - oy) + ox
 665            qy = sin(theta) * (px - ox) + cos(theta) * (py - oy) + oy
 666            coords[i] = qx
 667            coords[i + 1] = qy
 668        # set the coordinates:
 669        _a_canvas.coords(a_shape, coords)
 670
 671    return shape

 895def _interpolate_tuple(startcolor, goalcolor, frac):
 896    R = startcolor[0]
 897    G = startcolor[1]
 898    B = startcolor[2]
 899

 909    iG = int(G + (DiffG * frac))

 913    hG = hex(iG).replace("0x", "")
 914    hB = hex(iB).replace("0x", "")

 918    if len(hB) == 1:

 920    if len(hG) == 1:

 922
 923    color = ("#" + hR + hG + hB).upper()
 924
 925    return color
