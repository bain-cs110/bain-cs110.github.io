199def handle_typing(event):
200    '''
201    Event handler for key presses in W0rdle.
202
203    We need to handle 3 specific types of key presses:
204      1. `event.keysym == "BackSpace"`
205      2. `event.keysym == "Return"`
206      3. `len(event.keysym) == 1` (single character keys)
207
208    Args:
209      * `event`: The event to process.
210
211    Returns:
212      * `None` (Draws to the screen and also has side effects)
213    '''
214    # print("handle_typing called!")
215
216    # If the player hits BackSpace, delete the last character from the
217    # 'current_guess' key in the game_data dictionary
218    if event.keysym == "BackSpace":
219        game_data['current_guess'] = game_data['current_guess'][:-1]
220
221    # If the player hits Return...
222    elif event.keysym == "Return":
223        # First check that 5 letters have been entered
224        if len(game_data['current_guess']) != 5:
225            print("not enough letters")
226            # Next check if it's a valid word in our word list
227        elif game_data['current_guess'] not in game_data['word_list']:
228            print("not a valid word")
229        # If we make it past those two checks, it's a valid guess
230        else:
231            # First generate a hint, then finalize the guess
232            hint = generate_hint(game_data['current_guess'], game_data['solution'])
233            finalize_guess(game_data['current_guess'], hint)
234
235    # If the player hits any other letter/number/symbol on the keyboard
236    elif len(event.keysym) == 1:
237        # If the user hasn't entered 5 letters, add the entered symbol
238        # to the current guess (make sure to convert it to upper case!)
239        if len(game_data['current_guess']) < 5:
240            game_data['current_guess'] += event.keysym.upper()
241
242    # As long as the user hasn't made 6 guesses, show the game board
243    if len(game_data['past_guesses']) < 7:
244        show_game_board()
